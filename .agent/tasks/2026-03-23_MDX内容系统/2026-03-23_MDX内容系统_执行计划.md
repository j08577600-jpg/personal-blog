### Step 2: 重写 `src/lib/posts.ts`

**目标**: 将数据层从硬编码数组替换为文件系统读取

**文件**:
- `src/lib/posts.ts`（重写）

**操作**:

将 `src/lib/posts.ts` 整体替换为：

```ts
import fs from "fs";
import path from "path";
import matter from "gray-matter";
import readingTime from "reading-time";

const POSTS_DIR = path.join(process.cwd(), "content/posts");

export type Post = {
  slug: string;
  title: string;
  excerpt: string;
  date: string;
  readingTime: string;
  tags: string[];
  content: string;    // 原始 MDX 字符串
  published: boolean;
  cover?: string;
};

function parsePost(fileName: string): Post | null {
  const fullPath = path.join(POSTS_DIR, fileName);
  try {
    const fileContents = fs.readFileSync(fullPath, "utf8");
    const { data, content } = matter(fileContents);
    const rt = readingTime(content);

    return {
      slug: data.slug ?? fileName.replace(/\.mdx?$/, ""),
      title: data.title ?? "",
      excerpt: data.excerpt ?? "",
      date: data.date ?? "",
      tags: data.tags ?? [],
      published: data.published ?? false,
      cover: data.cover,
      readingTime: rt.text,
      content,
    };
  } catch {
    return null;
  }
}

export function getPosts(): Post[] {
  try {
    const fileNames = fs.readdirSync(POSTS_DIR);
    const posts = fileNames
      .filter((f) => f.endsWith(".mdx") || f.endsWith(".md"))
      .map(parsePost)
      .filter((p): p is Post => p !== null && p.published)
      .sort((a, b) => (a.date < b.date ? 1 : -1));
    return posts;
  } catch {
    return [];
  }
}

export function getPostBySlug(slug: string): Post | null {
  const candidates = [`${slug}.mdx`, `${slug}.md`];
  for (const name of candidates) {
    const post = parsePost(name);
    if (post) return post;
  }
  return null;
}
```

> 删除了 `posts` 数组和 `export const posts`。类型从 `content: string[]` 改为 `content: string`。

## 6. 决策日志

| 日期 | 决策 | 理由 |
|------|------|------|
| 2026-03-23 | `next-mdx-remote` RSC 导入路径 `from "next-mdx-remote/rsc"` | App Router Server Component 专用入口 |
| 2026-03-23 | 不加 `prose` class | `@tailwindcss/typography` 未安装，保守处理 |
| 2026-03-23 | 文件名匹配用 `slug.mdx` / `slug.md` 双扩展尝试 | frontmatter `slug` 字段可能与文件名不一致 |
| 2026-03-23 | `MDXRemote` 直接渲染，不包裹在 styled div 中 | `MDXRemote` 渲染自己的节点树，不能作为子元素放在 `space-y-6` div 内部 |

## 8. 恢复方式

| 场景 | 恢复方式 |
|------|----------|
| Build 失败 | 检查 TypeScript 类型，必要时 `// @ts-ignore` 临时跳过，或回退 `posts.ts` 到 `export const posts` 状态 |
| MDX 渲染失败 | 确认 `next-mdx-remote` 安装正确；检查 `MDXRemote` 导入路径是否为 `/rsc` |
| 列表页为空 | 检查 `content/posts/` 目录是否被 `.gitignore` 排除；确认 MDX 文件 `published: true` |
| 依赖冲突 | 使用 `npm ls gray-matter next-mdx-remote reading-time` 排查版本冲突 |
